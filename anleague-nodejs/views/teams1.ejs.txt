  <script>
    function toggleSquad(btn, index) {
      const squad = document.getElementById('squad-' + index);
      const isOpen = squad.classList.toggle('open');
      const playerCount = squad.querySelectorAll('.player').length;
      btn.textContent = isOpen ? 'Hide Squad' : `View Squad (${playerCount})`;
